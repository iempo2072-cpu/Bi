	if types == 1 then

local function GetDistanceFromStart(Objective, Start)
	if typeof(Objective) == "Vector3" and typeof(Start) == "Vector3" then
		return (Objective - Start).Magnitude
	elseif typeof(Objective) == "Vector2" and typeof(Start) == "Vector2" then

local function isBehindWall(startPos, endPos)
	if not startPos or not endPos then return false end
	local ray = Ray.new(startPos, endPos - startPos)
	local ignoreList = {}
	if Character then table.insert(ignoreList, Character) end
	local targets = c().TargetSlientAim
	if targets and targets.Character then table.insert(ignoreList, targets.Character) end
	local hit = workspace:FindPartOnRayWithIgnoreList(ray, ignoreList)
	return hit ~= nil
end;

local Hee = nil
local function Teleport(destination, value, t)
	c().StopWalking = false 
	c()['Running'] = true 
	local char = Client.Character or Client.CharacterAdded:Wait()
	local RootPart = char:WaitForChild("HumanoidRootPart")
	local Humanoid = char:WaitForChild("Humanoid")
	if Hee then Hee:Destroy() Hee = nil end
	local path = PathfindingService:CreatePath({
		AgentCanJump = true,
		AgentJumpHeight = 2.5,
		AgentHeight = 5.5,
		AgentRadius = 2.5,
		AgentMaxSlope = 90,
		Costs = {BlockedNode = 50, DoorArea = 1},
	})
	Hee = path 
	local success = pcall(function() path:ComputeAsync(RootPart.Position, destination) end)
	if not success then Hee = nil return end
	if path.Status == Enum.PathStatus.Success then
		local waypoints = path:GetWaypoints()
		for _, wp in pairs(waypoints) do
			if Hee ~= path then c()['Running'] = false return end
			local offsetY = (wp.Action == Enum.PathWaypointAction.Jump) and 10 or 1.5
			local targetPos = wp.Position + Vector3.new(0, offsetY, 0)
			local startPos = RootPart.Position
			local dir = (targetPos - startPos).Unit
			local dist = (targetPos - startPos).Magnitude
			local movedDist = 0
			local baseSpeed = 34
			local startTime = tick()
			Action(1,"set_sprinting_1", true)
			while movedDist < dist and value do
				task.wait()
				if Hee ~= path then c()['Running'] = false Action(1,"set_sprinting_1", false) return end
				if c().StopWalking or not value or not isAlive(Client) or Detect() then c()['Running'] = false break end
				if t and (CheckStatus(t) or (CheckingIsMinigame() and Teleport(destination, value, t))) then c()['Running'] = false break end
				local elapsedTime = tick() - startTime
				local dynamicSpeed = math.clamp(baseSpeed + (dist - movedDist) * 0.3, baseSpeed, 36)
				movedDist = math.min(elapsedTime * dynamicSpeed, dist)
				local newPos = startPos + dir * movedDist
				if Humanoid.Sit then Humanoid.Sit = false end

				Humanoid:Move(dir, false)
				RootPart.AssemblyLinearVelocity = dir * dynamicSpeed

				Character:PivotTo(CFrame.new(newPos))
			end
			Action(1,"set_sprinting_1", false)
			c()['Running'] = false
		end
	else
		Hee = nil
		c()['Running'] = false
	end
	if Hee == path then Hee = nil end
end

local function Drive(model, destination, value, t)
	c()["StopWalking"] = false
	if not model or not model.PrimaryPart then return end
	local path = PathfindingService:CreatePath({
		AgentRadius = c().VechineType == "car" and 5.5 or 2.750,
		AgentHeight = c().VechineType == "car" and 5.5 or 5,
		AgentCanJump = true,
		AgentMaxSlope = 45,
		Costs = { BlockedNode = 1000, Cars = 1 },
	})

	local success, err = pcall(function()
		path:ComputeAsync(model.PrimaryPart.Position, destination)
	end)

	if not success or path.Status ~= Enum.PathStatus.Success then c()['Running'] = false return end

	local waypoints = path:GetWaypoints()
	for i, wp in pairs(waypoints) do
		if not model or not model.PrimaryPart then return end

		local startPos = model:GetPivot().Position
		local goalPos = wp.Position + Vector3.new(0, 2, 0)
		local dir = (goalPos - startPos).Unit
		local dist = (goalPos - startPos).Magnitude
		local movedDist = 0
		local speed = c().VechineSpeed or 55
		local startTime = tick()

		while movedDist < dist and not c().StopWalking and value do task.wait()
			c()['Running'] = true
			if not isAlive(Client) then c()['Running'] = false c().StopWalking = true break end
			if not Humanoid.Sit then c()['Running'] = false c().StopWalking = true break end
			if CheckingIsMinigame() then c()['Running'] = false c().StopWalking = true break end
			if Detect() then c()['Running'] = false c().StopWalking = true Humanoid.Sit = false wait(.3) Drive(model, destination, value, t) break end
			if t and CheckStatus(t) then c()['Running'] = false c().StopWalking = true break end

			local dt = tick() - startTime
			movedDist = math.min(dt * speed, dist)
			local newPos = startPos + dir * movedDist

			if wp.Action == Enum.PathWaypointAction.Jump then 
				newPos = newPos + Vector3.new(0,12.5,0)
			end

			local Rotation = CFrame.lookAt(newPos, newPos + dir)

			model:PivotTo(Rotation)

			for _, part in pairs(model:GetDescendants()) do
				if part:IsA("BasePart") then
					part.AssemblyLinearVelocity = Vector3.zero
				end
			end
		end

		for _, part in pairs(model:GetDescendants()) do
			if part:IsA("BasePart") then
				part.AssemblyLinearVelocity = Vector3.zero
			end
		end
		c().StopWalking = false
		c()['Running'] = false
	end
end

local function ForceStop()
	if c()['Running'] then 
		c()["StopWalking"] = true 
		task.wait()
		if c()["StopWalking"] then  
			task.wait()
			c()["StopWalking"] = false
		end
	end
end

Func['EnabledSnap'] = function()
	local basePosition = RootPart.Position
	local lastOffset = c().YOffset or 1
	local Applies = 0 

	while task.wait() do
		if isAlive(Client) then
			if c().EnabledSnap then 
				local currentOffset = c().YOffset or 1

				if currentOffset ~= lastOffset then
					lastOffset = currentOffset
				end

				local currentY = RootPart.Position.Y
				local deltaY = GetDeltaY(basePosition.Y, currentY, currentOffset)

				RootPart.CFrame = RootPart.CFrame * CFrame.new(0, deltaY, 0)
				Applies = Applies + deltaY
			else
				if Applies ~= 0 then
					RootPart.CFrame = RootPart.CFrame * CFrame.new(0, Applies, 0)
					Applies = 0 
				end
			end
		end
	end
end

Func['AntiDied'] = function()

						local deltaY = GetDeltaY(RootPart.Position.Y, currentY, currentOffset)

						RootPart.Anchored = false 
						RootPart.CanCollide = false
						RootPart.CFrame = RootPart.CFrame * CFrame.new(0,deltaY,0)
						RootPart.Velocity = Vector3.new(RootPart.Velocity.X, -11, RootPart.Velocity.Z)

						for _, v in pairs(Character:GetDescendants()) do

								v.CanCollide = false 
								v.Anchored = false 
								v.CFrame = v.CFrame * CFrame.new(0,deltaY,0)

							end
						end
						if not EverDown then

						end
					end 
				else 
					EverDown = false
				end
