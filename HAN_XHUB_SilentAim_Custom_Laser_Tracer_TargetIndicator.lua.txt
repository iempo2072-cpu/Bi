local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local lp = Players.LocalPlayer
local playerGui = lp:WaitForChild("PlayerGui")

-- Helpers to safely call optional functions exported elsewhere
local function safeCall(fn, ...)
    if type(fn) ~= "function" then return false end
    local ok, res = pcall(fn, ...)
    return ok, res
end

-- Try to find module/global API from previous scripts
-- They might be globals: Config, separateModel, restoreModel, attachAntiKill, detachAntiKill
local DP_Config = rawget(_G, "DownedProtectConfig") or rawget(_G, "Config") or (rawget(_G, "_downedProtect") and rawget(_G, "_downedProtect").Config)
local separateModel = rawget(_G, "separateModel")
local restoreModel = rawget(_G, "restoreModel")
local attachAntiKill = rawget(_G, "attachAntiKill")
local detachAntiKill = rawget(_G, "detachAntiKill")

-- Fallback config if none exists
if not DP_Config then
    DP_Config = { Enabled = false, HealthTrigger = 30, AntiKillEnabled = true, StretchAmount = 2.8, SpreadRadius = 18 }
    rawset(_G, "DownedProtectConfig", DP_Config)
end

-- Utility to update UI visuals
local function updateUI()
    status.Text = "Status: " .. (DP_Config.Enabled and "ON" or "OFF")
    status.TextColor3 = DP_Config.Enabled and Color3.fromRGB(120,255,140) or Color3.fromRGB(255,120,120)
    btn.BackgroundColor3 = DP_Config.Enabled and Color3.fromRGB(34,139,34) or Color3.fromRGB(180,20,20)
    btn.Text = DP_Config.Enabled and "Disable Protection" or "Enable Protection"
end

-- Toggle action: set config, call attach/restore appropriately
local function setEnabled(enabled)
    DP_Config.Enabled = enabled
    -- If enabling and player currently downed or low health -> apply immediately
    local hum = lp.Character and lp.Character:FindFirstChildWhichIsA("Humanoid")
    if enabled then
        if hum then
            local downed = false
            pcall(function() downed = hum:GetAttribute and hum:GetAttribute("HasBeenDowned") end)
            if downed or (hum.Health and hum.Health <= (DP_Config.HealthTrigger or 30)) then
                pcall(function() separateModel(hum.Parent) end)
                if DP_Config.AntiKillEnabled then pcall(function() attachAntiKill(hum) end) end
            end
        end
    else
        -- disabling: restore visuals and detach anti-kill
        pcall(function() restoreModel() end)
        local hum2 = lp.Character and lp.Character:FindFirstChildWhichIsA("Humanoid")
        if hum2 then pcall(function() detachAntiKill(hum2) end) end
    end
    updateUI()
end

-- Button click
btn.MouseButton1Click:Connect(function()
    setEnabled(not DP_Config.Enabled)
end)

-- Allow draggable frame
local dragging, dragInput, dragStart, startPos
frame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position
        startPos = frame.Position
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)
frame.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement then
        dragInput = input
    end
end)
RunService.Heartbeat:Connect(function()
    if dragging and dragInput then
        local delta = dragInput.Position - dragStart
        frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)

-- Hotkey (optional): press RightShift to toggle
local UserInputService = game:GetService("UserInputService")
UserInputService.InputBegan:Connect(function(input, gpe)
    if gpe then return end
    if input.KeyCode == Enum.KeyCode.RightShift then
        setEnabled(not DP_Config.Enabled)
    end
end)

-- Keep UI in sync if other scripts change DP_Config.Enabled
RunService.Heartbeat:Connect(function()
    if DP_Config.Enabled ~= (status.Text:match("ON") and true or false) then
        updateUI()
    end
end)

-- Initial UI update
updateUI()
